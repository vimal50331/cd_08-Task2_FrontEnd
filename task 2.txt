    <script>
        function filterItems(category) {
            const items = document.querySelectorAll('li');
            items.forEach(item => {
                item.style.display = 'none';
            });
            if (category === 'all') {
                items.forEach(item => {
                    item.style.display = 'block';
                });
            } else {
                const filteredItems = document.querySelectorAll('.' + category);
                filteredItems.forEach(item => {
                    item.style.display = 'block';
                });
            }
        }

        function addItem() {
            const newItemInput = document.getElementById('new-item');
            const newItemText = newItemInput.value.trim();
            if (newItemText) {
                const category = determineCategory(newItemText);
                const groceryItems = document.getElementById('grocery-items');
                const newItem = document.createElement('li');
                newItem.className = category;
                newItem.textContent = newItemText;
                groceryItems.appendChild(newItem);
                newItemInput.value = '';
            }
        }

        function determineCategory(itemName) {
            // Implement logic to determine the category based on the item name.
            // You can use a simple approach like checking for specific keywords in the item name.
            // Example logic is just a placeholder.
            if (itemName.includes('Milk') || itemName.includes('Cheese') || itemName.includes('Yogurt') || itemName.includes('Butter') || itemName.includes('Eggs')) {
                return 'dairy';
            } else if (itemName.includes('Apples') || itemName.includes('Bananas') || itemName.includes('Oranges') || itemName.includes('Strawberries') || itemName.includes('Grapes') || itemName.includes('Pineapples')) {
                return 'fruits';
            } else if (itemName.includes('Carrots') || itemName.includes('Spinach') || itemName.includes('Bell Peppers') || itemName.includes('Cucumbers')) {
                return 'vegetables';
            } else {
                return 'other';
            }
        }
